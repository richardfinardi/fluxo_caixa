= 0 THEN 0 WHEN T3.CONCLUIDO = 1 THEN 1 WHEN SUM(ROUND(COALESCE(T1.hs_realizadas, 0), 2)) > SUM(ROUND(COALESCE(T1.total_hrs, 0), 2)) THEN 0.9 WHEN SUM(ROUND(COALESCE(T1.hs_realizadas, 0), 2)) = SUM(ROUND(COALESCE(T1.total_hrs, 0), 2)) THEN 1 ELSE 1 - COALESCE( CASE WHEN T3.CONCLUIDO = 1 THEN 0 WHEN SUM(ROUND(COALESCE(T1.total_hrs, 0), 2)) = 0 THEN 0 WHEN SUM(ROUND(COALESCE(T1.hs_realizadas, 0), 2)) > SUM(ROUND(COALESCE(T1.total_hrs, 0), 2)) THEN 0.1 ELSE ( SUM(COALESCE(T1.total_hrs, 0)) FILTER (WHERE T1.finalizado = 0) - COALESCE(SUM(T1.hs_realizadas) FILTER (WHERE T1.finalizado = 0), 0) ) / SUM(ROUND(COALESCE(T1.total_hrs, 0), 2)) END, 0) END AS perc_real, CASE WHEN COUNT(DISTINCT T1.CODIGO) = 0 THEN 'sem processos' WHEN COUNT(DISTINCT T1.CODIGO) FILTER ( WHERE T3.CODIGO, T3.n_os, T3.CONCLUIDO ), base_apontamentos AS ( SELECT CONCAT(T3.COD_EMPRESA, T3.CODIGO) AS id_os, MIN(T4.DATA_FIM) AS primeiro_apontamento, MAX(T4.DATA_FIM) AS ultimo_apontamento, T3.DT_CONCLUSAO FROM TOS T3 LEFT JOIN TOS_AUX T2 ON T3.COD_EMPRESA = T2.COD_EMPRESA AND T3.CODIGO = T2.COD_OS LEFT JOIN TPRO_PRO T1 ON T2.COD_EMPRESA = T1.COD_EMPRESA AND T2.COD_OS = T1.COD_OS AND T2.CODIGO = T1.COD_OS_AUX LEFT JOIN TCTRL_PH T4 ON T1.COD_EMPRESA = T4.COD_EMPRESA AND T1.CODIGO = T4.COD_BARR AND T1.COD_OS = T4.COD_OS AND T1.COD_OS_AUX = T4.COD_OS_AUX WHERE ( T1.COD_TP_SERVICO IS NULL OR T1.COD_TP_SERVICO NOT IN (10, 81, 82, 83, 84) ) AND ( T4.DATA_FIM IS NULL OR T4.DATA_FIM >= '20240101' ) GROUP BY T3.COD_EMPRESA, T3.CODIGO, T3.DT_CONCLUSAO ), base_ultimo_apontamento_engenharia AS ( SELECT DISTINCT ON (CONCAT(T3.COD_EMPRESA, T3.CODIGO)) CONCAT(T3.COD_EMPRESA, T3.CODIGO) AS id_os, T4.DATA_FIM AS data_ultimo_apontamento_eng, T4.FUNCIONARIO AS usuario_ultimo_apontamento_eng FROM TOS T3 INNER JOIN TOS_AUX T2 ON T3.COD_EMPRESA = T2.COD_EMPRESA AND T3.CODIGO = T2.COD_OS INNER JOIN TPRO_PRO T1 ON T2.COD_EMPRESA = T1.COD_EMPRESA AND T2.COD_OS = T1.COD_OS AND T2.CODIGO = T1.COD_OS_AUX INNER JOIN TCTRL_PH T4 ON T1.COD_EMPRESA = T4.COD_EMPRESA AND T1.CODIGO = T4.COD_BARR AND T1.COD_OS = T4.COD_OS AND T1.COD_OS_AUX = T4.COD_OS_AUX WHERE T1.COD_TP_SERVICO IN (10, 81, 82, 83, 84) AND T4.DATA_FIM IS NOT NULL ORDER BY CONCAT(T3.COD_EMPRESA, T3.CODIGO), T4.DATA_FIM DESC, T4.FIM DESC ) SELECT DISTINCT ON (tos.n_os) tos.dt_entrada, tos.n_os, tos.titulo, torcamento.n_orcamento, torcamento.versao, CASE -- quando veio de outro orçamento via M14 WHEN t_origem_orc.modulo_origem = 'M14' THEN ( SELECT torc.cliente FROM tos tos_rel LEFT torc ON tori_rel.cod_empresa = torc.cod_empresa AND tori_rel.cod_orcamento = torc.codigo WHERE tos_rel.n_os = t_origem_orc.n_servico LIMIT 1 ) -- origem normal ELSE torcamento.cliente END AS cliente,
